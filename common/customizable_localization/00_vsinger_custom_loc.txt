# This file adds Vsinger-specific titles to the customizable localization system.
# Using prepend_to_list ensures our definitions are checked before the default ones.

    name = GetWaifuType
    value = {
        text = {
            trigger = { is_vsinger_portrait_trigger = yes }
            localization_key = VSINGER_TYPE
        }
    }
}
