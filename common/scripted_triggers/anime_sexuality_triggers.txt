most_heterosexuality = {
	trigger_if = {
		limit = {
			waifu_portrait_trigger = no
		}
		has_game_rule = sexuality_distribution_default
	}
	trigger_else = {
		trigger_if = {
			limit = {
				is_female = yes
			}
			trigger_if = {
				limit = {
					has_game_rule = waifu_sexuality_distribution_default
				}
				has_game_rule = sexuality_distribution_default
			}
			trigger_else = {
				has_game_rule = waifu_sexuality_distribution_most_common_heterosexuality
			}
		}
		trigger_else = {
			trigger_if = {
				limit = {
					has_game_rule = husubando_sexuality_distribution_default
				}
				has_game_rule = sexuality_distribution_default
			}
			trigger_else = {
				has_game_rule = husubando_sexuality_distribution_most_common_heterosexuality
			}
		}
	}
}
